# c17 — ID Field (read-only, C-prefixed, compact) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Make the board table's `id` column auto-show each card's real ID, render it read-only-but-copyable in a compact column, and migrate the ID scheme to unpadded `C<n>` (uppercasing legacy `c<n>`).

**Architecture:** Three small, independent layers. (1) A canonical ID layer in `boardModel.ts` — pure helpers for extracting/normalizing/minting `C<n>` IDs, used by both the parser and the ops generator. (2) Parse-time migration that uppercases legacy `c<n>` IDs (and their body anchors) so they round-trip uppercased. (3) Render-time changes in `boardTableRender.ts` — a read-only branch for the `id` cell and a narrow default column width — plus a CSS rule for the muted look.

**Tech Stack:** TypeScript, Jest + ts-jest (jsdom for DOM tests), VS Code webview. Tests live in `tests/board/`. Run a single file with `npx jest <path>`.

```ts
/** Extract the trailing integer from a canonical `C<n>` or legacy `c<n>` id; null otherwise. */
export function idNumber(id: string): number | null {
  const m = /^[cC](\d+)$/.exec(id);
  return m ? parseInt(m[1], 10) : null;
}

/** Normalize a legacy lowercase `c<n>` id to the canonical uppercase `C<n>`. Idempotent. */
export function normalizeLegacyId(id: string): string {
  const m = /^c(\d+)$/.exec(id);
  return m ? `C${m[1]}` : id;
}

/** Next free id in the canonical `C<n>` scheme, continuing from the highest existing number. */
export function mintCardId(existingIds: Iterable<string>): string {
  const used = new Set<string>();
  let max = 0;
  for (const id of existingIds) {
    used.add(id);
    const n = idNumber(id);
    if (n !== null && n > max) max = n;
  }
  let n = max + 1;
  while (used.has(`C${n}`)) n++;
  return `C${n}`;
}
```

```ts
export function serializeBoard(board: Board): string {
  // De-duplicate card ids: first occurrence wins; later occurrences get -N suffix.
  // Empty ids are minted in the canonical C<n> scheme, continuing from the highest.
  let maxN = 0;
  for (const c of board.cards) {
    const n = idNumber(c.id);
    if (n !== null && n > maxN) maxN = n;
  }
  const seen = new Set<string>();
  const normalizedCards = board.cards.map((c) => {
    let id = c.id || `C${++maxN}`;
    if (!seen.has(id)) {
      seen.add(id);
      return { ...c, id, values: { ...c.values, id } };
    }
    let n = 2;
    while (seen.has(`${id}-${n}`)) n++;
    const next = `${id}-${n}`;
    seen.add(next);
    return { ...c, id: next, values: { ...c.values, id: next } };
  });
```
